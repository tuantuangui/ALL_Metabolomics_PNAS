library(dplyr)

dat <- data.table::fread("../raw_data/ALL-allsamples-pos.txt") %>%
  as.data.frame() %>%
  dplyr::mutate(component=toupper(`Component Name`)) %>%
  dplyr::mutate(library1=toupper(`Library Hit`)) %>%
  dplyr::filter(component==library1)
dat_rt <- dat %>%
  dplyr::group_by(`Component Name`,Formula) %>%
  dplyr::summarise(rt_median=60*median(`Retention Time`))
dat_theor <- dat %>%
  dplyr::group_by(`Component Name`) %>%
  dplyr::summarise(theor_median=median(`Precursor Mass`))
dat_found <- dat %>%
  dplyr::group_by(`Component Name`) %>%
  dplyr::summarise(found_median=median(`Found At Mass`))
dat_ppm <- dat %>%
  dplyr::group_by(`Component Name`) %>%
  dplyr::summarise(mass_error_median=median(`Mass Error (ppm)`))
dat_score <- dat %>%
  dplyr::group_by(`Component Name`) %>%
  dplyr::summarise(score_median=median(`Library Score`)/100)
  
dat_ab_positive <- dat_rt %>%
  dplyr::left_join(dat_theor,by="Component Name") %>%
  dplyr::left_join(dat_found,by="Component Name") %>%
  dplyr::left_join(dat_ppm,by="Component Name") %>%
  dplyr::left_join(dat_score,by="Component Name") %>%
  dplyr::mutate(`Component Name`=stringr::str_to_title(`Component Name`))


dat <- data.table::fread("result/dat_final.txt") %>%
  as.data.frame() %>%
  dplyr::filter(level=="AB")

positive_all <- data.table::fread("result/positive_all.txt") %>%
  as.data.frame() %>%
  rowwise() %>% 
  mutate(mean = mean(c_across(contains("QC")))) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::arrange(match(level,c("AB","metID","metDNA"))) %>%
  dplyr::distinct(refmet_name,.keep_all = TRUE) %>%
  dplyr::select(-mean) %>%
  dplyr::filter(level=="AB") %>%
  dplyr::select(level,label,refmet_name,formula)

dat_positive <- dat %>%
  dplyr::filter(ion=="positive") %>%
  dplyr::select(ion,label) %>%
  dplyr::left_join(positive_all,by=c("label"="refmet_name")) %>%
  dplyr::rename("metabolite name"="label.y") %>%
  dplyr::rename("refmet name"="label") %>%
  dplyr::select(ion,level,`metabolite name`,`refmet name`,formula)
dat_ab_positive_result <- dat_positive %>%
  dplyr::left_join(dat_ab_positive,by=c("metabolite name"="Component Name"))
write.table(dat_ab_positive_result,"result/metabolite_info/AB_positive_metabolite.txt",quote=F,row.names=F,sep="\t")
