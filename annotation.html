Chapter 9 annotation

9.1 metID

服务器上用metID对各个QC样本代谢物做鉴定，鉴定后对鉴定的代谢物做筛选

9.1.1 positive

library(dplyr)

file=list.files("positive/")

result <- data.table::fread(paste0("positive/",file[1]))
names(result)[2] <- file[1]

for (i in file[2:20]) {
  dat <- data.table::fread(paste0("positive/",i))
  names(dat)[2] <- i
  result <- result %>%
    dplyr::full_join(dat,by="name")
  
}

result$n <- rowSums(is.na(result))
result_filter <- result %>%
  dplyr::filter(n<19) %>%
  dplyr::select(-n)

n <- as.numeric()
name1 <- as.character()
for (i in 1:nrow(result_filter)) {
  n[i] <- length(as.character(na.omit(unique(as.character(result_filter[i,2:ncol(result_filter)])))))
  name1[i] <- paste(as.character(na.omit(unique(as.character(result_filter[i,2:ncol(result_filter)])))),collapse=";")
}
result_filter$n <- n
result_filter$name1 <- name1
result_filter1 <- result_filter %>%
  dplyr::mutate(result=ifelse(n==1,name1,NA)) %>%
  dplyr::mutate(result1=ifelse(n!=1,name1,NA))
write.table(result_filter1,"metID/positive_metID_v0920.txt",quote=F,row.names=F,sep="\t")

9.2 metDNA

线上用metDNA进行代谢物做鉴定，接着对鉴定到的代谢物做筛选

9.2.1 positive

library(dplyr)

metDNA_annotation <- read.csv("metDNA/positive_MRN.annotation.result.csv",header=T) %>%
  dplyr::filter(!is.na(compound.name)) %>%
  as.data.frame() %>%
  tidyr::separate(score,sep=";","Total.score") %>% ### if the compound.name have more than 1 name, then preserve the first one
  tidyr::separate(adduct,sep=";","new_adduct") %>%
  tidyr::separate(compound.name,sep=";","Compound.name")

#对metDNA对应到peak上

### if the compound.name have more than 1 name, then preserve the first one
metDNA_annotation_filter_temp <- metDNA_annotation %>%
  dplyr::filter(Total.score>0.7) %>%
  dplyr::filter(new_adduct %in% c("M+H","M+NH4")) 

## then unique the compound.name from the max score
metDNA_annotation_filter <- metDNA_annotation_filter_temp %>%
  as.data.frame() %>%
  dplyr::arrange(desc(Total.score)) %>%
  dplyr::distinct(Compound.name,.keep_all = TRUE) %>%
  dplyr::filter(!Compound.name=="NA") %>%
  dplyr::select(`Compound.name`,name)

write.table(metDNA_annotation_filter,"metDNA/positive_MRN.annotation.filter.txt",quote=F,sep="\t",row.names=F)

library(ggplot2)

positive_metID <- data.table::fread("metID/positive_metID_v0920.txt") %>%
  as.data.frame() %>%
#  dplyr::filter(!is.na(result_name)) %>%

  dplyr::mutate(sum=sum(`2QC10-Sample167`,`2QC11-Sample177`)) %>%
  dplyr::arrange(desc(sum)) %>%
  dplyr::select(-sum) %>%
  dplyr::distinct(`Compound.name`,.keep_all = TRUE) %>%
  dplyr::select(name,mz,rt,Compound.name,level,everything())

write.table(dat,"result/2.positive_data_xcms_filter.txt",quote=F,row.names=F,sep="\t")

library(dplyr)

dat <- data.table::fread("result/2.positive_data_xcms_filter.txt") %>%
