##impute, QC1 using QC1 metabolites' mean, QC2 using QC2 metabolites' mean, other using 0.1*min in every sample
QC1_impute <- t(apply(QC1,1,function(x){Hmisc::impute(x,mean)}))
QC2_impute <- t(apply(QC2,1,function(x){Hmisc::impute(x,mean)}))
sample_impute <- apply(sample,2,function(x){x[which(is.na(x))] <- 0.1*min(x,na.rm=T);return(x)})
