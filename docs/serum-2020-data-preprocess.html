library(dplyr)

path="/Users/guituantuan/Desktop/projects/junyu/raw_data/serum_2020/positive/"
file <- list.files(path)

mydata1 <- read.table(paste0(path,file[1]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
  tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
  tidyr::separate("s1",c(NA,"B")) %>%
  dplyr::mutate(sample_name=paste0("id_",B)) %>%
  dplyr::select(-"B") %>%
  tibble::column_to_rownames("sample_name") %>%
  t() %>%
  as.data.frame() %>%
  dplyr::rename(Metabolites=id_NA)

mydata2 <- read.table(paste0(path,file[2]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
  tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
  tidyr::separate("s1",c(NA,"B")) %>%
  dplyr::mutate(sample_name=paste0("id_",B)) %>%
  dplyr::select(-"B") %>%
  tibble::column_to_rownames("sample_name") %>%
  t() %>%
  as.data.frame() %>%
  dplyr::rename(Metabolites=id_NA)

result <- rbind(mydata1,mydata2)
for (i in 3:length(file)){
  mydata <- read.table(paste0(path,file[i]),sep="\t",header=F,check.names = F) %>%
    dplyr::select(-c("V1","V3")) %>%
    tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
    tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
    tidyr::separate("s1",c(NA,"B")) %>%
    dplyr::mutate(sample_name=paste0("id_",B)) %>%
    dplyr::select(-"B") %>%
    tibble::column_to_rownames("sample_name") %>%
    t() %>%
    as.data.frame() %>%
    dplyr::rename(Metabolites=id_NA)
  result <- rbind(result,mydata)
}
rownames(result) <- NULL

result1 <- result %>%
  dplyr::mutate(mean=apply(result[,2:ncol(result)],1,function(x){return(mean(as.numeric(x),na.rm=T))})) %>%

  dplyr::distinct(Metabolites,.keep_all=T) %>%
  dplyr::select(-"mean") %>%
  tibble::column_to_rownames("Metabolites") %>%
  dplyr::mutate_if(is.character, as.numeric)

result1[result1==0] <- NA
