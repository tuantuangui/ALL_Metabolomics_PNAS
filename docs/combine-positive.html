positive_all <- data.table::fread("result/positive_all.txt") %>%
  as.data.frame() %>%
  rowwise() %>% 
  mutate(mean = mean(c_across(contains("QC")))) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::arrange(match(level,c("AB","metID","metDNA"))) %>%
  dplyr::distinct(refmet_name,.keep_all = TRUE) %>%
  dplyr::select(-mean) %>%
  dplyr::select(level,refmet_name)

positive <- data.table::fread("result/positive_after_SERRF.txt") %>%
  as.data.frame() %>%
  dplyr::mutate(ion="positive") %>%
  dplyr::left_join(positive_all,by=c("label"="refmet_name")) %>%
  dplyr::select(level,ion,everything())

negative_all <- data.table::fread("result/negative_all.txt") %>%
  as.data.frame() %>%
  rowwise() %>% 
  mutate(mean = mean(c_across(contains("QC")))) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::arrange(match(level,c("AB","metID","metDNA"))) %>%
  dplyr::distinct(refmet_name,.keep_all = TRUE) %>%
  dplyr::select(-mean) %>%
  dplyr::select(level,refmet_name)
negative <- data.table::fread("result/negative_after_SERRF.txt") %>%
  as.data.frame() %>%
  dplyr::mutate(ion="negative") %>%
  dplyr::left_join(negative_all,by=c("label"="refmet_name")) %>%
  dplyr::select(level,ion,everything())

sample_overlap <- intersect(names(positive),names(negative))

dat <- rbind(positive %>% dplyr::select(all_of(sample_overlap)),negative %>% dplyr::select(all_of(sample_overlap)))

dat_filter <- dat %>%
  rowwise() %>% 
  mutate(mean = mean(c_across(contains("QC")))) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::arrange(match(level,c("AB","metID","metDNA"))) %>%
  dplyr::distinct(label,.keep_all = TRUE) %>%
  dplyr::filter(!label %in% c("3',5' cyclic AMP","beta-Tyrosine")) %>%
  dplyr::select(-mean)
write.table(dat_filter,"result/dat_final.txt",quote=F,row.names=F,sep="\t")
