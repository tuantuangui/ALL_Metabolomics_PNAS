2.1 serum 2020 positive data preprocess

2.1.1 合并2020年serum的positive数据及对na值的填充

library(dplyr)

path="/Users/guituantuan/Desktop/projects/junyu/raw_data/serum_2020/positive/"
file <- list.files(path)

mydata1 <- read.table(paste0(path,file[1]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
  tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
  tidyr::separate("s1",c(NA,"B")) %>%
  dplyr::mutate(sample_name=paste0("id_",B)) %>%
  dplyr::select(-"B") %>%
  tibble::column_to_rownames("sample_name") %>%
  t() %>%
  as.data.frame() %>%
  dplyr::rename(Metabolites=id_NA)

mydata2 <- read.table(paste0(path,file[2]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
  tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
  tidyr::separate("s1",c(NA,"B")) %>%
  dplyr::mutate(sample_name=paste0("id_",B)) %>%
  dplyr::select(-"B") %>%
  tibble::column_to_rownames("sample_name") %>%
  t() %>%
  as.data.frame() %>%
  dplyr::rename(Metabolites=id_NA)

result <- rbind(mydata1,mydata2)
for (i in 3:length(file)){
  mydata <- read.table(paste0(path,file[i]),sep="\t",header=F,check.names = F) %>%
    dplyr::select(-c("V1","V3")) %>%
    tidyr::separate("V2",sep=",",c("sample_name",NA)) %>%
    tidyr::separate("sample_name",sep="POS",c(NA,"s1")) %>%
    tidyr::separate("s1",c(NA,"B")) %>%
    dplyr::mutate(sample_name=paste0("id_",B)) %>%
    dplyr::select(-"B") %>%
    tibble::column_to_rownames("sample_name") %>%
    t() %>%
    as.data.frame() %>%
    dplyr::rename(Metabolites=id_NA)
  result <- rbind(result,mydata)
}
rownames(result) <- NULL

result1 <- result %>%
  dplyr::mutate(mean=apply(result[,2:ncol(result)],1,function(x){return(mean(as.numeric(x),na.rm=T))})) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::distinct(Metabolites,.keep_all=T) %>%
  dplyr::select(-"mean") %>%
  tibble::column_to_rownames("Metabolites") %>%
  dplyr::mutate_if(is.character, as.numeric)

result1[result1==0] <- NA

source("/Users/guituantuan/Desktop/projects/script/impute.R")

result_impute <- meta_impute(result1) %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="Metabolites")

library(dplyr)

path="/Users/guituantuan/Desktop/projects/junyu/raw_data/serum_2020/negative/"

file <- list.files(path)

mydata1 <- read.table(paste0(path,file[1]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tibble::column_to_rownames("V2")

mydata2 <- read.table(paste0(path,file[2]),sep="\t",header=F,check.names = F) %>%
  dplyr::select(-c("V1","V3")) %>%
  tibble::column_to_rownames("V2")

result <- cbind(mydata1,mydata2)
for (i in 3:length(file)){
  mydata <- read.table(paste0(path,file[i]),sep="\t",header=F,check.names = F) %>%
    dplyr::select(-c("V1","V3")) %>%
    tibble::column_to_rownames("V2")
  result <- cbind(result,mydata)
}

result_temp <- result %>%
  tibble::rownames_to_column(var="V1") %>%
  tidyr::separate("V1",sep=",",c("sample_name",NA)) %>%
  tidyr::separate("sample_name",sep="NEG",c(NA,"s1")) %>%
  tidyr::separate("s1",c(NA,"B")) %>%
  dplyr::mutate(sample_name=paste0("id_",B)) %>%
  dplyr::select(-"B")

result_temp$mean <- apply(result_temp[,-which(names(result_temp)=="sample_name")],1,function(x){return(mean(as.numeric(x),na.rm=T))})

result_temp_sample_mean <- result_temp %>%
  dplyr::arrange(desc(as.numeric(mean))) %>%
  dplyr::distinct(sample_name,.keep_all=TRUE) %>%
  dplyr::select(-"mean") %>%
  tibble::column_to_rownames("sample_name") %>%
  t() %>%
  as.data.frame()
result_temp_sample_mean$mean <- apply(result_temp_sample_mean[,-which(names(result_temp_sample_mean)=="id_NA")],1,function(x){return(mean(as.numeric(x),na.rm=T))})

rownames(result_temp_sample_mean) <- NULL
result_temp_metabolties_mean <- result_temp_sample_mean %>%
  dplyr::arrange(desc(as.numeric(mean))) %>%
  dplyr::distinct(id_NA,.keep_all=TRUE) %>%
  dplyr::select(-"mean") %>%
  tibble::column_to_rownames("id_NA") %>%
  dplyr::mutate_if(is.character, as.numeric)

result_temp_metabolties_mean[result_temp_metabolties_mean==0] <- NA

result1 <- result_temp_metabolties_mean

  as.data.frame() %>%
  tibble::rownames_to_column(var="Metabolites")

#添加内标
