10.1 combine the data from AB and xcms

10.1.1 positive

输出结果后需要人工对refmet name做修正,修正后的文件名为result/positive_all.txt

library(dplyr)

data_ab_positive <- data.table::fread("result/1.AB_positive_name2refmet_library_result.txt") %>%
  dplyr::filter(filter!="del") %>%
  as.data.frame() %>%
  dplyr::select(-`Library Hit`,-filter) %>%
  dplyr::mutate(level="AB") %>%
  dplyr::select(level,everything())

data_metid_positive <- data.table::fread("result/2.positive_xcms_name2refmet.txt") %>%
  as.data.frame() %>%
  dplyr::select(-name,-mz,-rt) %>%
  dplyr::rename("label"="Compound.name") %>%
  dplyr::select(level,everything()) %>%
  t() %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="a") %>%
  tidyr::separate(a,sep="-","a") %>%
  tibble::column_to_rownames("a") %>%
  t() %>%
  as.data.frame()
overlap_sample <- intersect(names(data_ab_positive),names(data_metid_positive))
data_metid_positive_filter <- data_metid_positive %>%
  dplyr::select(overlap_sample) %>%
  dplyr::filter(!refmet_name %in% data_ab_positive$refmet_name)

data_ab_positive_filter <- data_ab_positive %>%
  dplyr::select(overlap_sample)

dat <- rbind(data_ab_positive_filter,data_metid_positive_filter) %>%
  unique()

write.table(dat,"result/positive_all_temp.txt",quote=F,row.names=F,sep="\t")

10.2 transform the format to serrf

10.2.1 positive

dat <- data.table::fread("result/positive_all.txt") %>%
  as.data.frame() %>%
  rowwise() %>% 
  mutate(mean = mean(c_across(contains("QC")))) %>%
  dplyr::arrange(desc(mean)) %>%
  dplyr::arrange(match(level,c("AB","metID","metDNA"))) %>%
  dplyr::distinct(label,.keep_all = TRUE) %>%
  dplyr::select(-mean)

order_intersect <- data.table::fread("raw_data/positive_qc_intersect.txt") %>%
  as.data.frame() %>%
  dplyr::filter(!sample_id %in% c("QC11","QC12","2QC3","2QC6","2QC9","2QC12")) %>%
  dplyr::mutate(type=ifelse(type=="QC","qc",type))

#把数据调成serrf所需要的样子
all_for_serrf <- dat %>%
  dplyr::select(-level,-label,-refmet_name_raw,-formula,-super_class,-main_class,-sub_class) %>%
  dplyr::rename("label"="refmet_name")

overlap_sample <- intersect(names(all_for_serrf),order_intersect$sample_id)

qc_intersect_filter <- order_intersect %>%
  dplyr::filter(sample_id %in% overlap_sample) %>%
  dplyr::arrange(time)
qc_intersect_filter$time <- seq(1,nrow(qc_intersect_filter))

all_for_serrf_filter <- all_for_serrf %>%
  dplyr::select(c("label",qc_intersect_filter$sample_id))

batch <- c("batch",qc_intersect_filter$batch)
type <- c("sampleType",qc_intersect_filter$type)
time <- c("time",qc_intersect_filter$time)
label <- c(colnames(all_for_serrf_filter))

all_result_temp <- rbind(batch,type,time,label,all_for_serrf_filter)

number <- c("","","","No",c(1:(nrow(all_for_serrf_filter))))
serrf_data <- cbind(number,all_result_temp)

write.table(serrf_data,"result/positive_for_serrf.txt",quote=F,row.names=F,sep="\t")
