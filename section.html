library(dplyr)
library(ggplot2)

before_normalization_raw <- data.table::fread("result/positive_for_serrf.txt") %>%
  as.data.frame()

before_normalization <- before_normalization_raw %>%
  t() %>%
  as.data.frame() %>%
  filter(!V4 %in% c("No","label"))

names(before_normalization) <- as.character(before_normalization_raw[,2])
names(before_normalization)[4] <- "sample_id"

# pca
before_batch <- as.character(before_normalization_raw[1,-c(1,2)])
before_type <- as.character(before_normalization_raw[2,-c(1,2)])
before_sample_id <- as.character(before_normalization_raw[4,-c(1,2)])

dd <- data.frame(type=before_type,batch=before_batch,sample_id=before_sample_id)

mydata_pca <- apply(before_normalization[,5:ncol(before_normalization)],2,as.numeric)

rownames(mydata_pca) <- before_sample_id
aa_positive <- cbind(before_type,before_batch,before_sample_id) %>%
  as.data.frame()
mydata_pca_positive <- mydata_pca %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sampleid")


### negative
before_normalization_raw <- data.table::fread("result/negative_for_serrf.txt") %>%
  as.data.frame()

before_normalization <- before_normalization_raw %>%
  t() %>%
  as.data.frame() %>%
  filter(!V4 %in% c("No","label"))

names(before_normalization) <- as.character(before_normalization_raw[,2])
names(before_normalization)[4] <- "sample_id"

# pca
before_batch <- as.character(before_normalization_raw[1,-c(1,2)])
before_type <- as.character(before_normalization_raw[2,-c(1,2)])
before_sample_id <- as.character(before_normalization_raw[4,-c(1,2)])

dd <- data.frame(type=before_type,batch=before_batch,sample_id=before_sample_id)

mydata_pca_negative <- apply(before_normalization[,5:ncol(before_normalization)],2,as.numeric)

rownames(mydata_pca_negative) <- before_sample_id
mydata_pca_negative <- mydata_pca_negative %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sampleid")
aa_negative <- cbind(before_type,before_batch,before_sample_id) %>%
  as.data.frame()

result <- mydata_pca_positive %>%
  dplyr::inner_join(mydata_pca_negative,by="sampleid") %>%
  tibble::column_to_rownames("sampleid")

aa <- aa_positive %>%
  dplyr::inner_join(aa_negative,by="before_sample_id")

pca <- prcomp(log2(result+1), center = T, scale. = T)
variance = pca$sdev^2/sum(pca$sdev^2)

pca.data = data.frame(pca$x,type=aa$`before_type.x`,batch=aa$before_batch.x,sample_id=aa$before_sample_id)

#col <- hue_pal()(4)

p1 <- ggplot(pca.data,aes(PC1,PC2,color=type,shape=batch))+
  geom_point()+
  scale_color_manual(values=c("red","#3e68a0"))+
  scale_shape_manual(values=seq(0,length(unique(aa$before_batch.x))-1))+
  #geom_text(aes(label=sample_id))+
  theme_bw()+
  theme(panel.grid.major = element_blank(),panel.grid.minor = element_blank())+
  labs(x=paste0("PC1 (",signif(variance[1]*100,3),"%)"),
       y=paste0("PC2 (",signif(variance[2]*100,3),"%)"))

library(dplyr)
library(ggplot2)

after_normalization_raw <- read.csv("result/SERRF/positive_normalized_SERRF.csv") %>%
  as.data.frame() %>%
  dplyr::distinct(label,.keep_all = T) %>%
  tibble::column_to_rownames("label") %>%
  t() %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sample") %>%
  dplyr::mutate(sample=gsub("X","",sample)) %>%
  tibble::column_to_rownames("sample")

after_sample_id <- rownames(after_normalization_raw)
after_dd <- data.frame(sample_id=after_sample_id) %>%
  dplyr::left_join(dd,by="sample_id")

mydata_pca <- apply(after_normalization_raw,2,as.numeric)
rownames(mydata_pca) <- after_dd$sample_id
mydata_pca <- mydata_pca %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sampleid")


after_normalization_raw_negative <- read.csv("result/SERRF/negative_normalized_SERRF.csv") %>%
  as.data.frame() %>%
  dplyr::distinct(label,.keep_all = T) %>%
  tibble::column_to_rownames("label") %>%
  t() %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sample") %>%
  dplyr::mutate(sample=gsub("X","",sample)) %>%
  tibble::column_to_rownames("sample")

after_sample_id_negative <- rownames(after_normalization_raw_negative)

after_dd_negative <- data.frame(sample_id=after_sample_id_negative) %>%
  dplyr::left_join(dd,by="sample_id")

mydata_pca_negative <- apply(after_normalization_raw_negative,2,as.numeric)
rownames(mydata_pca_negative) <- after_dd_negative$sample_id
mydata_pca_negative <- mydata_pca_negative %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="sampleid")


result <- mydata_pca %>%
  dplyr::inner_join(mydata_pca_negative,by="sampleid") %>%
  tibble::column_to_rownames("sampleid")

ss <- after_dd_negative %>%
  dplyr::filter(sample_id %in% rownames(result)) %>%
  dplyr::arrange(match(sample_id,rownames(result)))

pca <- prcomp(log2(result), center = T, scale. = T)
variance = pca$sdev^2/sum(pca$sdev^2)

pca.data = cbind(pca$x,ss)

p2 <- ggplot(pca.data,aes(PC1,PC2,color=type,shape=batch))+
  geom_point()+
  scale_color_manual(values=c("red","#3e68a0"))+
  scale_shape_manual(values=seq(0,1))+
  #geom_text(aes(label=sample_id))+
  theme_bw()+
  theme(panel.grid.major = element_blank(),panel.grid.minor = element_blank())+
  labs(x=paste0("PC1 (",signif(variance[1]*100,3),"%)"),
       y=paste0("PC2 (",signif(variance[2]*100,3),"%)"))

ggsave("result/QC/3.pca_normalization.pdf",p2,width=5,height=4)

library(ggplot2)

dat_positive_rsd<- data.table::fread("result/SERRF/positive_QC-RSDs.csv") %>%
  as.data.frame() %>%
  dplyr::filter(SERRF<= 0.3)

dat_negative_rsd <- data.table::fread("result/SERRF/negative_QC-RSDs.csv") %>%
  as.data.frame() %>%
  dplyr::filter(SERRF <= 0.3)


dat <- rbind(dat_positive_rsd,dat_negative_rsd)


dat_melt <- reshape2::melt(dat,id="label") %>%
  dplyr::mutate(variable=ifelse(variable=="none","Raw","Normalization")) %>%
  dplyr::mutate(variable=factor(variable,levels = c("Raw","Normalization")))

p <- ggplot(dat_melt,aes(x=variable,y=value))+
  geom_violin(col="#D01910",width=.9)+
  geom_jitter(width=.2,col="#00599F",size=0.5)+
  theme_bw()+
  labs(x=NULL,y="RSD")
ggsave("result/QC/3.RSD.pdf",p,width=3.5,height = 3)
