8.1 filter the metabolites (too much NA) and impute the NA

8.1.1 positive

library(dplyr)
library(MNet)

dat <- data.table::fread("raw_data/matric table-ALL-allsamples-POS.txt",skip=1) %>%
  dplyr::filter(!`Original Filename` %in% "Sample Type") %>%
  tibble::column_to_rownames("Original Filename") %>%
  t() %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="Original Filename") %>%
  dplyr::mutate(`Original Filename`=gsub("0911\\\\pos\\\\QC","0911\\\\pos\\\\2QC",`Original Filename`)) %>%
  tidyr::separate(`Original Filename`,sep="pos",c(NA,"n")) %>%
  tidyr::separate(n,sep="\\\\",c(NA,"n")) %>%
  tidyr::separate(n,sep="[.]","Sample Name") %>%
  dplyr::distinct(`Sample Name`,.keep_all = T) %>%
  tibble::remove_rownames() %>%
  tibble::column_to_rownames("Sample Name") %>%
  t() %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="label") %>%
  unique() %>%
  dplyr::mutate(label=stringr::str_to_title(label)) %>%
  tibble::column_to_rownames("label") %>%
  dplyr::mutate_if(is.character,as.numeric)

n <- rowSums(is.na(dat))

dat$n <- n

dat_filter <- dat %>% 
  dplyr::filter(n<32.6) %>%
  dplyr::select(-n)

dat_impute <- meta_impute(dat_filter,method="min") %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="label")

write.table(dat_impute,"result/1.AB_positive_impute.txt",quote=F,row.names=F,sep="\t")

library(dplyr)

result <- dat %>%

library(dplyr)

dat_library <- data.table::fread("raw_data/ALL-allsamples-pos.txt") %>%
  dplyr::select(`Component Name`,`Library Hit`) %>%
  dplyr::mutate(`Component Name`=stringr::str_to_title(`Component Name`)) %>%
  unique() %>%
  dplyr::group_by(`Component Name`) %>%
  dplyr::summarise(paste(`Library Hit`,collapse = ";"))

names(dat_library)[2]="Library Hit"

result <- data.table::fread("result/1.AB_positive_name2refmet.txt") %>%
  dplyr::left_join(dat_library,by=c("label"="Component Name")) %>%
  dplyr::select(label,`Library Hit`,everything())
