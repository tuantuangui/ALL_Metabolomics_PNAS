args=commandArgs(T)

library(dplyr)
library(data.table)

setwd(args[1])

##read the negative
negative <- data.table::fread("negative_for_serrf.txt") %>%
  as.data.frame()


####negative level info
negative_level_raw <- data.table::fread("temp/negative_level_anno.txt") %>%
  as.data.frame()

##read the positive
positive <- data.table::fread("positive_for_serrf.txt") %>%
  as.data.frame()


####positive level info
positive_level_raw <- data.table::fread("temp/positive_level_anno.txt") %>%
  as.data.frame()

negative_level <- negative_level_raw %>%
  dplyr::select(c("refmet_name","level"))
names(negative_level)[1] <- "label"

negative_qc <- negative[-c(1,2,3,4),c(2,which(negative[2,]=="QC"))]
negative_sample <- negative[-c(1,2,3,4),c(2,which(negative[2,]=="sample"))]

negative_rsd <- apply(negative_qc[,-1],1,function(x){sd(as.numeric(x),na.rm=T)/mean(as.numeric(x),na.rm=T)})
negative_sample_mean <- apply(negative_sample[,-1],1,function(x){mean(as.numeric(x),na.rm=T)})

negative$V1 <- c("NA","NA","NA","rsd",negative_rsd)
negative$sample_mean <- c("NA","NA","NA","sample_mean",negative_sample_mean)

colnames(negative) <- as.character(negative[4,])

negative_merge <- merge(negative,negative_level,by="label",all.x=T,sort=F) %>%
  dplyr::arrange(match(label,negative$label)) %>%
  dplyr::filter(!label %in% c("batch","type","time","label"))

negative_filter <- negative_merge
#negative_filter <- negative_merge %>%
#  dplyr::filter(rsd<0.3)

positive_level <- positive_level_raw %>%
  dplyr::select(c("refmet_name","level"))
names(positive_level)[1] <- "label"

positive_qc <- positive[-c(1,2,3,4),c(2,which(positive[2,]=="QC"))]
positive_sample <- positive[-c(1,2,3,4),c(2,which(positive[2,]=="sample"))]
positive_rsd <- apply(positive_qc[,-1],1,function(x){sd(as.numeric(x),na.rm=T)/mean(as.numeric(x),na.rm=T)})
positive_sample_mean <- apply(positive_sample[,-1],1,function(x){mean(as.numeric(x),na.rm=T)})

positive$V1 <- c("NA","NA","NA","rsd",positive_rsd)
positive$sample_mean <- c("NA","NA","NA","sample_mean",positive_sample_mean)

colnames(positive) <- as.character(positive[4,])

positive_merge <- merge(positive,positive_level,by="label",all.x=T,sort=F) %>%
  dplyr::arrange(match(label,positive$label)) %>%
  dplyr::filter(!label %in% c("batch","type","time","label"))

#positive_filter <- positive_merge %>%
#  dplyr::filter(rsd<0.3)

positive_filter <- positive_merge

##get the positive and negative overlap sample
positive_negative_intersect_id <- intersect(colnames(positive_filter),colnames(negative_filter))


positive_intersect <- positive_filter %>%
  dplyr::select(all_of(positive_negative_intersect_id))
positive_intersect <- cbind(type="positive",positive_intersect)

negative_intersect <- negative_filter %>%
  dplyr::select(all_of(positive_negative_intersect_id))
negative_intersect <- cbind(type="negative",negative_intersect)

positive_negative_combine <- rbind(positive_intersect,negative_intersect) %>%
  dplyr::select(!"rsd")

##unique the metabolite and get the max
positive_negative_combine_temp <- as.data.frame(apply(positive_negative_combine[,-c(1,2)],2,as.numeric))
positive_negative_combine_temp$label <- positive_negative_combine$label
positive_negative_combine_temp$type <- positive_negative_combine$type

####留下小level的代谢物及相同level留下sample值最大的模式数据
positive_negative_combine_uniq <- positive_negative_combine_temp %>%
  dplyr::arrange(desc(sample_mean)) %>%
  dplyr::arrange(level) %>%
  dplyr::distinct(label,.keep_all=TRUE)%>%
  dplyr::select(!"sample_mean")

all_level <- rbind(positive_level_raw,negative_level_raw) %>%
  dplyr::select(-"level") %>%
  unique()

all_result <- merge(all_level,positive_negative_combine_uniq,by.y="label",by.x="refmet_name",all.y=T)

all_result <- all_result %>%
  dplyr::select(c("type","level",names(all_result))) %>%
  dplyr::arrange(level)

write.table(all_result,"3.positive_negative_combine_uniq_raw.txt",sep="\t",quote=F,row.names=F,col.names=T)
