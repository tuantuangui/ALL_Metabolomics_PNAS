Chapter 7 batch effection of data from serum 2020 and serum 2021 using sva

Rscript 55-combine_2020_2021_raw_sva_v0222.R result/v0224_serum_2020/ result/v0224_serum_2021/ result/v0224_serum_combine_2020_2021/

args=commandArgs(T)

library(dplyr)


serum_2020 <- data.table::fread(paste0(args[1],"/3.positive_negative_combine_uniq_raw.txt")) %>%
  as.data.frame()

serum_2021 <- data.table::fread(paste0(args[2],"/3.positive_negative_combine_uniq_raw.txt")) %>%
  as.data.frame()

###取出2020和2021 overlap的代谢物
metabolites_2020_2021_overlap <- intersect(serum_2020$refmet_name,serum_2021$refmet_name)

###筛选出2020的代谢物
serum_2020_filter <- serum_2020 %>%
  dplyr::filter(refmet_name %in% metabolites_2020_2021_overlap) %>%
  dplyr::select(-c("type","level","formula","exactmass","super_class","main_class","sub_class","class_index"))

###筛选出2021的代谢物
serum_2021_filter <- serum_2021 %>%
  dplyr::filter(refmet_name %in% metabolites_2020_2021_overlap) %>%
  dplyr::select(-c("type","level","formula","exactmass","super_class","main_class","sub_class","class_index"))

###对2020的样本名称前加y2020_,对2021的样本名称前加y2021_
names(serum_2020_filter) <- paste0("y2020_",names(serum_2020_filter))
names(serum_2021_filter) <- paste0("y2021_",names(serum_2021_filter))
names(serum_2020_filter)[1] <- "refmet_name"
names(serum_2021_filter)[1] <- "refmet_name"



clinical_2020 <- data.table::fread(paste0(args[1],"/positive_qc_intersect.txt")) %>%
  as.data.frame() %>%
  dplyr::mutate(sample_id=paste0("y2020_",sample_id)) %>%
  dplyr::arrange(time) %>%
  dplyr::filter(sample_id %in% names(serum_2020_filter)) %>%
  dplyr::select(c("sample_id","batch","type","group"))

clinical_2021 <- data.table::fread(paste0(args[2],"/positive_qc_intersect.txt")) %>%
  as.data.frame() %>%
  dplyr::mutate(sample_id=paste0("y2021_",sample_id)) %>%
  dplyr::arrange(time) %>%
  dplyr::filter(sample_id %in% names(serum_2021_filter)) %>%
  dplyr::mutate(batch=ifelse(batch=="F","J",
                             ifelse(batch=="E","I",
                                    ifelse(batch=="D","H",
                                           ifelse(batch=="C","G",
                                                  ifelse(batch=="B","F",
                                                         ifelse(batch=="A","E",NA))))))) %>%
  dplyr::mutate(type=ifelse(type1=="QC2","QC","sample")) %>%
  dplyr::select(c("sample_id","batch","type","group"))

clinical_all <- rbind(clinical_2020,clinical_2021) %>%
  dplyr::mutate(group=ifelse(is.na(group),"QC",group))
clinical_all$time <- seq(1,nrow(clinical_all))

###合并2020和2021数据

all_data <- merge(serum_2020_filter,serum_2021_filter,by="refmet_name") %>%
  dplyr::select(c("refmet_name",clinical_all$sample_id)) %>%
  tibble::column_to_rownames("refmet_name")


clinical_all$group <- as.factor(clinical_all$group)
model <- model.matrix(~group, data = clinical_all)

combat_edata <- sva::ComBat(dat = as.matrix(log2(all_data)), batch = clinical_all$batch,mod = model)
dat <- 2** combat_edata

clinical_all1 <- clinical_all %>%
  dplyr::select(c("batch","type","time","sample_id")) %>%
  dplyr::rename("label"="sample_id") %>%
  t()
all_dat <- rbind(clinical_all1,dat) %>%
  as.data.frame() %>%
  tibble::rownames_to_column(var="a")
v1 <- c("","","","No",seq(1,nrow(all_dat)-4))
all_dat_need <- cbind(v1=v1,all_dat)

write.table(all_dat_need,paste0(args[3],"/serum_raw_sva_for_serrf.txt"),quote=F,sep="\t",row.names = F,col.names=F)
